---
title: Components
description: Compound components for declarative text layout
---

react-pretext provides a compound component API via the `Pretext` namespace. Import it as:

```tsx
import { Pretext } from "react-pretext";
```

Individual components are also available as named exports: `PretextProvider`, `PretextRoot`, `PretextLines`, `PretextLine`.

---

## Pretext.Provider

Provides default `font`, `lineHeight`, `locale`, and `debug` values to all descendant components and hooks.

```tsx
<Pretext.Provider font="16px Inter" lineHeight={24}>
  {children}
</Pretext.Provider>
```

<ParamField body="font" type="string" required>
  CSS font shorthand (e.g. `"16px Inter"`, `"bold 14px monospace"`)
</ParamField>

<ParamField body="lineHeight" type="number" required>
  Line height in pixels
</ParamField>

<ParamField body="locale" type="string">
  Locale for text segmentation (e.g. `"en"`, `"ja"`)
</ParamField>

<ParamField body="debug" type="boolean">
  Enable debug overlays on `Pretext.Line` components
</ParamField>

---

## Pretext.Root

Computes text layout and makes it available to children. Supports two patterns:

### Child component pattern

Children access layout through `Pretext.Lines`:

```tsx
<Pretext.Root text="Hello world" width={300}>
  <Pretext.Lines>
    {(line) => <div key={line.key}>{line.text}</div>}
  </Pretext.Lines>
</Pretext.Root>
```

### Render prop pattern

Full control over the layout result:

```tsx
<Pretext.Root text={text} width={300}>
  {(layout) => (
    <svg height={layout.height}>
      {layout.lines.map((line) => (
        <text key={line.key} y={line.y}>{line.text}</text>
      ))}
    </svg>
  )}
</Pretext.Root>
```

<ParamField body="text" type="string" required>
  Text content to lay out
</ParamField>

<ParamField body="width" type="number" required>
  Maximum width in pixels for line breaking
</ParamField>

<ParamField body="font" type="string">
  Overrides the font from `Pretext.Provider`
</ParamField>

<ParamField body="lineHeight" type="number">
  Overrides the lineHeight from `Pretext.Provider`
</ParamField>

<ParamField body="locale" type="string">
  Locale for text segmentation
</ParamField>

<ParamField body="children" type="ReactNode | (layout: PretextLayoutResult) => ReactNode" required>
  Either React nodes (use with `Pretext.Lines`) or a render function receiving the layout result
</ParamField>

---

## Pretext.Lines

Iterates over lines from the nearest `Pretext.Root`. Must be a descendant of `Pretext.Root`.

```tsx
<Pretext.Lines>
  {(line, index) => (
    <div key={line.key} style={{ position: "absolute", top: line.y }}>
      {line.text}
    </div>
  )}
</Pretext.Lines>
```

<ParamField body="children" type="(line: PretextLineInfo, index: number) => ReactNode" required>
  Render function called for each line. Receives the line info and its index.
</ParamField>

---

## Pretext.Line

Single-line wrapper with optional debug overlay that shows line width measurements.

```tsx
<Pretext.Lines>
  {(line) => (
    <Pretext.Line key={line.key} line={line} debug>
      {(line) => <span>{line.text}</span>}
    </Pretext.Line>
  )}
</Pretext.Lines>
```

<ParamField body="line" type="PretextLineInfo" required>
  The line info object from `Pretext.Lines` or a layout result
</ParamField>

<ParamField body="children" type="ReactNode | (line: PretextLineInfo) => ReactNode">
  Content to render. Can be a render function or static nodes. Defaults to `line.text` if omitted.
</ParamField>

<ParamField body="debug" type="boolean">
  Show a red outline and width measurement overlay. Falls back to the `debug` value from `Pretext.Provider`.
</ParamField>

---

## Example: Full Component Tree

```tsx
import { Pretext } from "react-pretext";

function Article({ title, body }: { title: string; body: string }) {
  return (
    <Pretext.Provider font="16px Inter" lineHeight={24}>
      <div style={{ maxWidth: 600, margin: "0 auto" }}>
        <h1>{title}</h1>
        <Pretext.Root text={body} width={600}>
          <Pretext.Lines>
            {(line) => (
              <Pretext.Line key={line.key} line={line}>
                {(l) => (
                  <div style={{ position: "absolute", top: l.y }}>
                    {l.text}
                  </div>
                )}
              </Pretext.Line>
            )}
          </Pretext.Lines>
        </Pretext.Root>
      </div>
    </Pretext.Provider>
  );
}
```
