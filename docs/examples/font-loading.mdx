---
title: Font Loading
description: Show a loading state while a web font loads, then re-layout
---

Use `useFontReady` to detect when a web font finishes loading, then re-layout with exact measurements. This prevents layout shift when switching from a fallback font to the target font.

## Key Concepts

- `useFontReady` returns `false` until the font is loaded, then `true`
- Use a fallback font for initial render, swap to the target font when ready
- Layout automatically recomputes when the font prop changes

## Code

```tsx
import { useFontReady, usePretextLayout } from "react-pretext";

function FontAwareText({ text }: { text: string }) {
  const font = "16px 'Playfair Display'";
  const fallbackFont = "16px serif";
  const ready = useFontReady(font);

  const layout = usePretextLayout({
    text,
    maxWidth: 400,
    font: ready ? font : fallbackFont,
    lineHeight: 28,
  });

  return (
    <div style={{ position: "relative", height: layout.height }}>
      {!ready && (
        <div style={{ position: "absolute", top: 0, right: 0, fontSize: 12, color: "#999" }}>
          Loading font...
        </div>
      )}
      {layout.lines.map((line) => (
        <div
          key={line.key}
          style={{
            position: "absolute",
            top: line.y,
            whiteSpace: "pre",
            font: ready ? font : fallbackFont,
            transition: "opacity 0.2s",
            opacity: ready ? 1 : 0.7,
          }}
        >
          {line.text}
        </div>
      ))}
    </div>
  );
}
```
