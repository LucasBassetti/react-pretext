title: Responsive Layout
description: Automatically re-layout text when the container resizes

Automatically re-layout text when the container resizes by combining `useResizeObserver` with `Pretext.Root`. The layout recomputes on every width change — since `layout()` is pure arithmetic (~0.0002ms), no debouncing is needed.

## Key Concepts

- `useResizeObserver` tracks the container's content box dimensions
- `useFontReady` detects when a web font finishes loading
- Layout recomputation is effectively free for stable inputs thanks to LRU caching

## Code

```tsx
import { useRef } from "react";
import {
  Pretext,
  useResizeObserver,
  useFontReady,
} from "react-pretext";

function ResponsiveArticle({ text }: { text: string }) {
  const containerRef = useRef<HTMLDivElement>(null);
  const { width } = useResizeObserver(containerRef);
  const fontReady = useFontReady("16px Inter");

  return (
    <div ref={containerRef} style={{ width: "100%", maxWidth: 800, margin: "0 auto" }}>
      {width > 0 && (
        <Pretext.Root text={text} width={width} font="16px Inter" lineHeight={24}>
          {(layout) => (
            <div style={{ position: "relative", height: layout.height }}>
              <Pretext.Lines>
                {(line) => (
                  <div
                    key={line.key}
                    style={{
                      position: "absolute",
                      top: line.y,
                      left: 0,
                      whiteSpace: "pre",
                      opacity: fontReady ? 1 : 0.5,
                    }}
                  >
                    {line.text}
                  </div>
                )}
              </Pretext.Lines>
            </div>
          )}
        </Pretext.Root>
      )}
    </div>
  );
}
```
